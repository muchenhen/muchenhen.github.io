用OpenGL绘制两个相连三角形 2019-03-01|OpenGL

源码：

#include #include #include void framebuffer_size_callback(GLFWwindow* window, int width, int height);
void processInput(GLFWwindow *window);

// 设置窗口大小
const unsigned int SCR_WIDTH = 800;
const unsigned int SCR_HEIGHT = 600;

//编写顶点着色器
const char *vertexShaderSource = “#version 460 core\n”
“layout (location = 0) in vec3 aPos;\n”
“void main()\n”
“{\n”
“ gl_Position = vec4(aPos.x, aPos.y, aPos.z, 1.0);\n”
“}\0”;

//编写片段着色器
const char *fragmentShaderSource = “#version 460 core\n”
“out vec4 FragColor;\n”
“void main()\n”
“{\n”
“ FragColor = vec4(1.0f, 0.5f, 0.2f, 1.0f);\n”
“}\n\0”;

int main()
{
glfwInit();
glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

//创建窗口
GLFWwindow\* window = glfwCreateWindow(SCR\_WIDTH, SCR\_HEIGHT, "相连的三角行", NULL, NULL);
//检查窗口创建
if (window == NULL)
{
    std::cout << "Failed to create GLFW window" << std::endl;
    glfwTerminate();
    return -1;
}
glfwMakeContextCurrent(window);
glfwSetFramebufferSizeCallback(window, framebuffer\_size\_callback);

if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress))
{
    std::cout << "Failed to initialize GLAD" << std::endl;
    return -1;
}

// 创建并编译着色器
// 顶点着色器
int vertexShader = glCreateShader(GL\_VERTEX\_SHADER);
glShaderSource(vertexShader, 1, &vertexShaderSource, NULL);
glCompileShader(vertexShader);
// 检查编译错误
int success;
char infoLog\[512\];
glGetShaderiv(vertexShader, GL\_COMPILE\_STATUS, &success);
if (!success)
{
    glGetShaderInfoLog(vertexShader, 512, NULL, infoLog);
    std::cout << "ERROR::SHADER::VERTEX::COMPILATION\_FAILED\\n" << infoLog << std::endl;
}
// 片段着色器
int fragmentShader = glCreateShader(GL\_FRAGMENT\_SHADER);
glShaderSource(fragmentShader, 1, &fragmentShaderSource, NULL);
glCompileShader(fragmentShader);
//检查编译错误
glGetShaderiv(fragmentShader, GL\_COMPILE\_STATUS, &success);
if (!success)
{
    glGetShaderInfoLog(fragmentShader, 512, NULL, infoLog);
    std::cout << "ERROR::SHADER::FRAGMENT::COMPILATION\_FAILED\\n" << infoLog << std::endl;
}
//链接着色器
int shaderProgram = glCreateProgram();
glAttachShader(shaderProgram, vertexShader);
glAttachShader(shaderProgram, fragmentShader);
glLinkProgram(shaderProgram);
//检查连接错误
glGetProgramiv(shaderProgram, GL\_LINK\_STATUS, &success);
if (!success) {
    glGetProgramInfoLog(shaderProgram, 512, NULL, infoLog);
    std::cout << "ERROR::SHADER::PROGRAM::LINKING\_FAILED\\n" << infoLog << std::endl;
}
glDeleteShader(vertexShader);
glDeleteShader(fragmentShader);

// 设置顶点数据和缓存，配置顶点属性
// 添加一个新的顶点数组，形成两个三角形，共6个顶点
float vertices\[\] = {
    //第一个三角形
    -0.9f, -0.5f, 0.0f,  // left 
    -0.0f, -0.5f, 0.0f,  // right
    -0.45f, 0.5f, 0.0f,  // top 
    //第二个三角形
     0.0f, -0.5f, 0.0f,  // left
     0.9f, -0.5f, 0.0f,  // right
     0.45f, 0.5f, 0.0f   // top 
};

unsigned int VBO, VAO;
glGenVertexArrays(1, &VAO);
glGenBuffers(1, &VBO);
//绑定顶点数组对象
glBindVertexArray(VAO);
//绑定并设置顶点缓存
glBindBuffer(GL\_ARRAY\_BUFFER, VBO);
glBufferData(GL\_ARRAY\_BUFFER, sizeof(vertices), vertices, GL\_STATIC\_DRAW);
//配置顶点属性
glVertexAttribPointer(0, 3, GL\_FLOAT, GL\_FALSE, 3 \* sizeof(float), (void\*)0);
glEnableVertexAttribArray(0);

glBindBuffer(GL\_ARRAY\_BUFFER, 0);

glBindVertexArray(0);

//渲染循环
while (!glfwWindowShouldClose(window))
{
    //输入
    processInput(window);

    //渲染
    glClearColor(0.2f, 0.3f, 0.3f, 1.0f);
    glClear(GL\_COLOR\_BUFFER\_BIT);

    //画图
    glUseProgram(shaderProgram);
    glBindVertexArray(VAO);
    glDrawArrays(GL\_TRIANGLES, 0, 6); //两个三角形共六个顶点

    //交换缓冲区
    glfwSwapBuffers(window);
    glfwPollEvents();
}

//取消用过的资源的分配
glDeleteVertexArrays(1, &VAO);
glDeleteBuffers(1, &VBO);

// glfw终止，清除所有分配的资源
glfwTerminate();
return 0;

}

//检测并响应按键输入
void processInput(GLFWwindow *window)
{
if (glfwGetKey(window, GLFW_KEY_ESCAPE) == GLFW_PRESS)
glfwSetWindowShouldClose(window, true);
}

//改变窗口大小时调用
void framebuffer_size_callback(GLFWwindow* window, int width, int height)
{
glViewport(0, 0, width, height);
}

OpenGL渲染管线概要 2019-03-01|计算机图形学OpenGL

渲染管线概要

  • 顶点数据 Vertex Data 一系列顶点的集合

    • 顶点 Vertex 一个3D坐标的数据的集合
    • 顶点属性 Vertex Atrribute 用来表示顶点数据，例如一个3D位置和一些颜色值

  • 顶点输入

    • 数组 保存每一个顶点的3D位置

    • 标准化设备坐标 Normalized Device Coordinates 3个轴为-1.0到1.0，坐标值必须在此范围之内

    • 顶点着色器 在GPU上创建内存储存顶点数据，解释这些内存并指定如何发送给显卡，接着会处理在内存中指定数量的顶点。

    • 顶点缓冲对象 Vertex Buffer Objects 管理VS创建的内存，在显存中储存大量顶点。

      • glGenBuffers函数 使用glGenBuffers函数和一个缓冲ID生成一个VBO对象： unsigned int VBO; glGenBuffers(1, &VBO);

      • 顶点缓冲类型 GL_ARRAY_BUFFER

      • glBindBuffer函数 绑定不同类型的缓冲，把新创建的缓冲绑定到GL_ARRAY_BUFFER目标上 glBindBuffer(GL_ARRAY_BUFFER, VBO); ​

      • glBufferData函数 把之前定义的顶点数据复制到缓冲的内存中： glBufferData(GL_ARRAY_BUFFER,sizeof(vertices),vertices,GL_STATIC_DRAW);

        • 目标的缓冲类型

        • 指定传输数据大小，单位为字节

        • 希望发送的实际数据

        • 指定显卡如何管理给定数据

          • GL_STATIC_DRAW 数据不会或几乎不会改变
          • GL_DYNAMIC_DRAW 数据会被改变很多
          • GL_STREAM_DRAW 数据每次绘制时都会改变

  • 图元 Primitive 指定数据所表示的渲染类型

    • GL_POINTS 一系列点
    • GL_TRIANGLES 一系列三角形
    • GL_LINE_STRIP 一系列连续直线
    • GL_LINES 每一对顶点被解释为一条直线

  • 渲染管线 Graphics Pipeline

    • 顶点着色器 Vertex Shader 把3D坐标转为另一种3D坐标 允许对顶点属性进行一些基本处理​

      • 编写

      • 编译

        • glCreateShader函数 创建着色器，将着色器类型以参数形式传给函数

        • glShaderSource函数

          • 要编译的着色器对象
          • 指定传递的源码字符串数量
          • 顶点着色器的真正的源码
          • NULL

        • glCompileShader函数

    • 图元装配 Primitive Assembly 将所有点装配成指定图元

    • 几何着色器 Geometry Shader 输入：图元形式的一系列顶点集合 输出：通过产生新顶点构造出新的或是其他的图元生成其他形状​

    • 光栅化 Rasterization Stage 将图元映射为屏幕上的像素，生成片段

    • 裁切 Clipping 丢弃视图之外的像素

    • 片段着色器 Fragment Shader 计算一个像素的最终颜色，通常包含3D场景的数据，如光照、阴影、光的颜色

      • 片段 Fragment 光栅化生成

      • 输出变量 输出一个四分量向量，表示颜色的RGBA值

      • 编译

        • glCreatShader
        • glShaderSource
        • glCompileShader

    • Alpha测试与混合 Blending 深度值判断消隐，并根据透明度进行混合​

  • 着色器程序 着色器程序对象Shader Program Object是多个着色器合并之后并最终链接完成的版本

    • 创建一个程序对象 unsigned int shaderProgram;​

      • glCreateProgram函数 创建一个程序并返回新创建程序对象的ID引用

      • glAttachShader函数

        • 程序对象
        • 着色器对象

      • glLinkProgram函数 得到的结果是一个程序对象

      • glUseProgram函数 激活这个程序对象，该函数调用之后，每一个着色器调用和渲染调用都会使用这个程序对象

      • glDeleteShader函数 着色器对象链接到程序对象后删除着色器对象

  • 顶点属性

    • 顶点数据缓冲

      • 紧密排列 Tightly Packed

    • glVertexAttribPointer函数 每个顶点属性从一个VBO管理的内存中获得它的数据，而具体是从哪个VBO（程序中可以有多个VBO）获取则是通过在调用glVetexAttribPointer时绑定到GL_ARRAY_BUFFER的VBO决定的。由于在调用glVetexAttribPointer之前绑定的是先前定义的VBO对象，顶点属性0现在会链接到它的顶点数据。

      • 配置的顶点属性 在定义顶点着色器中定义的position顶点属性的位置值Location
      • 顶点属性的大小 三维的vec3则由三个值组成，大小是3
      • 数据类型 如GL_FLOAT
      • 是否标准化 GL_TRUE所有数据会被映射到0到1，signed为-1到1
      • 步长 Stride 连续顶点属性组之间的间隔。当数组为紧密排列时可以设置为0让OpenGL决定步长
      • 偏移量 Offset 参数类型为 void*，表示位置数据在缓冲中起始位置的偏移量

    • glEnableVertexAttribArray函数 启用顶点属性

  • 顶点数组对象 Vertex Array Object 可以像顶点缓冲对象那样被绑定，任何随后的顶点属性调用都会储存在这个VAO中。这样的好处就是，当配置顶点属性指针时，你只需要将那些调用执行一次，之后再绘制物体的时候只需要绑定相应的VAO就行了。这使在不同顶点数据和属性配置之间切换变得非常简单，只需要绑定不同的VAO就行了。刚刚设置的所有状态都将存储在VAO中。

    • 创建

      • glGenVertexArrays函数

    • 绑定VAO

      • glBindVertexArray函数

    • 顶点数组复制

      • glBindBuffer函数
      • glBufferData函数

    • 设置顶点属性指针

      • glVertexAttribPointer函数
      • glEnableVertexAttribArray函数

    • 绘制物体

      • glUseProgram函数
      • glBindVertexArray函数
      • glDrawArrays绘制函数 使用当前激活的着色器，之前定义的顶点属性配置和VBO的顶点数据来绘制图元

  • 索引缓冲对象 Elment Buffer Object 也叫做Index Buffer Object​ EBO是一个专门存储所以的缓冲，OpenGL调用这些顶点的索引来决定该绘制哪个顶点。​

    • 创建

    • 绑定

    • 顶点数组复制

    • 索引数组复制到索引缓冲

      • glBindBuffer函数
      • glBufferData函数

    • 设置顶点属性指针

    • 绘制

      • glDrawElements函数
      • glBindVertexArray函数

对美好事物追忆的使用 2019-01-03|惘

This article was translated from the economist on December 29, 2018.

————————————————————

如何从回忆思潮的爆发中获得最好的事物

政客们总是利用过去。而现在，发达国家和新兴的经济体们正在经历一场回忆的爆发。左派和右派，民主和专制都在追忆着昔日的荣光。正当美国总统唐纳德特朗普发誓要“让美国再次伟大”的时候，中国国家主席习近平正在使用他的“中国梦”去驱散百年的屈辱，让中国回到她的黄金时代。墨西哥新任总统安德烈斯·曼纽尔·洛佩兹·奥夫拉多尔肩负着抵御全球资本主义、恢复国家经济主权的使命。波兰最有权势的政治家雅罗斯瓦夫•卡钦斯基希望清除苏联共产主义的最后一丝痕迹，实现波兰传统价值观的复兴。

这场回忆的狂欢，在不同的国家有着不同的原因。在新兴的市场，过去的辉煌往往意味着未来的胜利。中国已经经历了40年的转型增长，她意识到自己正站在伟大事业的起点上。印度在纳伦德拉•莫迪的领导下，一直在用印度教民族主义复兴来庆祝其日益增长的地缘政治影响力。

相比之下，在发达国家，怀旧通常源自亨利•杰克逊协会的索菲娅•加斯顿所称的“无所不在的、具有威胁性的衰退感觉”。近三分之二的英国人认为过去的生活更好。同样比例的法国人也觉得现在的生活不舒服。今年联合国世界幸福报告中，美国民众的幸福感在降低。无论是在发达国家还是发展中国家，绝大多数的人认为机器人和自动化将会加剧不公平的现象，并且伤害到就业。在一项2017年针对28个国家所做的一项调查报告中现实，超过半数的人认为，自己的生活在将来不会有任何改善，甚至可能会更差。只有不到15%的日本民众认为他们的孩子将来会比他们的父母更加富有。

大量的研究结果证明了，新兴市场的乐观主义比起发达国家的悲观主义，是建立在更坚实的基础上的。世界平均的人类寿命正变得越来越长，人们的生活也变得更加健康；战争和饥荒带走的生命越来越少；教育的普及也使得歧视与偏见慢慢减少。但是同样的，夏天不像人们想象的那样美好，国家也不像人们想象的那么光荣。

但是仅仅因为悲观主义和怀旧情绪的不准确而拒绝它们，就大错特错。它们是塑造政治的强大力量。要想驾驭它们，首先要理解他们。

怀旧对于乐观主义者和悲观主义者都是改变世界的支柱。包括人工智能在内的各种新技术，有可能改变整个实体工业，甚至影响国家和公民的关系。两个世纪以来，世界的中心力量正在从西方转移回中国。地球老化的速度比历史上任何时期都要快。她的气候在改变，而生活在她上面的种族和文化变得更加矛盾与复杂。

在这样的时刻，人们会被怀旧所吸引，把它作为一种安慰和自尊的来源。许多脱欧派人士认为离开欧盟意味着他们将再次加入一个充满活力的全球化的英国；加泰罗尼亚人在追求独特身份的过程中唤起了一段理想化的过去；巴西人选出了一位让人回想起三十年前他们摆脱的军事独裁统治的总统。当特朗普推动煤炭和钢铁行业发展时，那些担心自己在肮脏、垂死的行业中被边缘化的人，突然觉得自己又有了价值。

在发达国家，怀旧也提供了一种方式，来反抗别人的进步观念，从而夺回控制权。在前东德，极右翼政党——德国另类选择党拥有最强大的支持，那里的选民对失去社区和安全感到遗憾。

在法国，“黄背心”砸碎了香榭丽舍大街上的商店橱窗，因为他们入不敷出。他们拒绝他们的总统伊曼纽尔·马克龙提出的国家繁荣与个人经济安全之间的权衡。

这使得怀旧变得危险。抗议者并不一定希望时光倒流，他们可能只是想让时间慢下来。然而，这种僵化可能只会加剧衰退的感觉。此外，怀旧者渴望的自尊似乎常常助长仇外情绪。在印度，印度教民族主义复兴的过程中，针对穆斯林的仇恨犯罪有所增加。在西方，就在强硬的左翼人士谴责全球商业的阴谋诡计时，右翼人士记得一个更“白”、文化更少的过去。反犹太主义卷土重来并非偶然，尤其是在态度保守的杰里米•科尔宾领导的英国工党内部。

由于仇外心理的影响，进步人士很快就把所有的怀旧情绪视为偏见，导致他们忽视了整个社会阶层的恐惧。这种笼统的判断是民粹主义者能够如此成功地利用怀旧情绪的又一个原因。

他们过得太安逸了。怀旧可以被永久地利用。20世纪初，同样受到技术、地缘政治和文化变革的冲击，欧洲和美国也很怀旧。随后，一段时期的冲突和社会动荡导致了普选和全民教育。今天的政客可以从那个时代吸取教训。显然，它们必须通过维护和加强使各国能够合作的机构来避免战争。但他们也应该找到大胆的方法来应对不安全感和疏离感。这意味着国家机器需要为公民更加努力地工作，使教育贯穿公民的一生，改革税收，将权力下放，应对气候灾难，以及明智地管理移民。

如果“过去”是“外国”，让我移民吧。

怀旧者们有他们的道理。当一种管理世界的方式似乎已经用尽力量，而另一种方式还没有出现时，从过去吸取教训显得尤为重要。当一切似乎都没有意义时，历史就成了最高的学科。知道你是谁，从哪里来，这很重要。利用过去的最好方法是破除偏见，开拓视野。正确的历史观会帮助你明白，进步取决于面对艰难的选择。有时它也能激励人。50年前，阿波罗8号从佛罗里达州卡纳维拉尔角起飞。12月24日，它拍下了一幅地球的照片，这是一颗半遮半掩的蓝白色行星，看起来很团结。
